516 	/**
517 	 * Dispatch Event
518 	 *
519 	 * @param	{String}	eventType	Type of event
520 	 * @param	{String}	data	Data passed as arguments to the functions that are listening
521 	 *
522 	 * @public
523 	 */
524 	function dispatchEvent(eventType, data) {
525 		$(document).trigger(eventType, data);
526 	}
527 	
528 	/**
529 	 * Listen for event
530 	 *
531 	 * @param	{String}	eventType	Type of event you are listening to
532 	 * @param	{String}	func	Function that will be triggered when the event is fired
533 	 *
534 	 * @public
535 	 */
536 	function addEventListener(eventType, func) {
537 		$(document).bind(eventType, function(e, data) { func(data); });
538 	}
539 	
540 	/**
541 	 * Dispatch Event to Flash Object. 
542 	 *
543 	 * @example	<br />Usage: dispatchEvent("flashFunction", parameter1, parameter2);
544 	 *
545 	 * @param	{String}	func	Function to call in flash. Arguments can be passed as well as extra parameters
546 	 *
547 	 * @public
548 	 */
549 	function dispatchFlashEvent(func) {
550 		if(_oFlash && _isFlashReady) {
551 			if(arguments.length > 1)
552 				_oFlash[func](Array.prototype.slice.call(arguments).slice(1)[0]);
553 			else
554 				_oFlash[func]();
555 		}
556 	}
