550 	/**
551 	 * Dispatch Event
552 	 *
553 	 * @example	<br />Usage: dispatchEvent("flashFunction", parameter);
554 	 *
555 	 * @param	{String}	eventType	Type of event
556 	 * @param	{String}	data	Data passed as arguments to the functions that are listening
557 	 *
558 	 * @public
559 	 */
560 	function dispatchEvent(eventType, data) {
561 		dispatchFlashEvent("handleJSCall", eventType, data);
562 		
563 		$(document).trigger(eventType, data);
564 	}
565 	
566 	/**
567 	 * Listen for event
568 	 *
569 	 * @param	{String}	eventType	Type of event you are listening to
570 	 * @param	{String}	func	Function that will be triggered when the event is fired
571 	 *
572 	 * @public
573 	 */
574 	function addEventListener(eventType, func) {
575 		$(document).bind(eventType, function(e, data) { func(data); });
576 	}
577 	
578 	/**
579 	 * Dispatch Event to Flash Object. 
580 	 *
581 	 * @example	<br />Usage: dispatchFlashEvent("flashFunction", parameter1, parameter2, ...);
582 	 *
583 	 * @param	{String}	func	Function to call in flash. Arguments can be passed as well as extra parameters
584 	 *
585 	 * @public
586 	 */
587 	function dispatchFlashEvent(func) {
588 		var args = Array.prototype.slice.call(arguments).slice(1);
589 		
590 		trace("DFE: " + JSON.stringify(args));
591 		
592 		if(_oFlash && _isFlashReady) {
593 			if(arguments.length > 1)
594 				_oFlash[func](args);
595 			else
596 				_oFlash[func]();
597 		}
598 	}

661 function IFrame(parentElement)
662 {
663    // Create the iframe which will be returned
664    var iframe = document.createElement("iframe");
665 
666    // If no parent element is specified then use body as the parent element
667    if(parentElement == null)
668       parentElement = document.body;
669 
670    // This is necessary in order to initialize the document inside the iframe
671    parentElement.appendChild(iframe);
672 
673    // Initiate the iframe's document to null
674    iframe.doc = null;
675 
676    // Depending on browser platform get the iframe's document, this is only
677    // available if the iframe has already been appended to an element which
678    // has been added to the document
679    if(iframe.contentDocument)
680       // Firefox, Opera
681       iframe.doc = iframe.contentDocument;
682    else if(iframe.contentWindow)
683       // Internet Explorer
684       iframe.doc = iframe.contentWindow.document;
685    else if(iframe.document)
686       // Others?
687       iframe.doc = iframe.document;
688 
689    // If we did not succeed in finding the document then throw an exception
690    if(iframe.doc == null)
691       throw "Document not found, append the parent element to the DOM before creating the IFrame";
692 
693    // Create the script inside the iframe's document which will call the
694    iframe.doc.open();
695    iframe.doc.close();
696 
697    // Return the iframe, now with an extra property iframe.doc containing the
698    // iframe's document
699    return iframe;
700 }
